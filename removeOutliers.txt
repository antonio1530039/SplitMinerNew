makeChanges(LinkedHashMap<HashMap<ArrayList<ArrayList<Character>>, SignificantContext> significantContexts, LinkedHashMap<Integer, ArrayList<Character>> traces){

		for(sc : significantContexts){
			if(sc exists in t){
				t.replace(sc.bestCover);
			}
		}
	}
